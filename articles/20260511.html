题目：西交大校赛

C

取y为x的二进制最小1,此时y能整除x，x&y=y，gcd（x，y）=y

// 代码示例
void solve()
{
    ll x; cin >> x;
    ll p = x;
    ll ans = 1;
    ll o = 0;
    while(p > 0)
    {
        if(p % 2 == 0)
        {
            ans += pow(2, o);
        }
        if(gcd(ans, x) == (ans & x)) break;
        o++;
        p /= 2;
    }
    cout << ans << endl;
}

D

bfs遍历每一个格子，对每个格子的4个方向扩展，判断这个格子所在连通块是否同色

// 代码示例
int dx[4]={0,0,-1,1};
int dy[4]={1,-1,0,0};
int n,m;
int ok = 1;


void bfs(vector<string>&s,vector<vector<int>>&vis)
{
    for(int i = 0; i < n; i++)
    {
        for(int j = 0; j < m; j++)
        {
            queue<pair<int, int>>q;
            if(!vis[i][j])
            {
                q.push({i,j});
                vis[i][j] = 1;
                char color = s[i][j];
                int min_r = i, max_r = i;
                int min_c = j, max_c = j;

                while(!q.empty())
                {
                    auto [x,y] = q.front();
                    q.pop();

                    min_r = min(min_r,x);
                    max_r = max(max_r,x);
                    min_c = min(min_c,y);
                    max_c = max(max_c,y);

                    for(int k = 0; k < 4; k++)
                    {
                        int nx = x + dx[k];
                        int ny = y + dy[k];

                        if(nx >= 0 && nx < n && ny >= 0 && ny < m && !vis[nx][ny] && s[nx][ny] == color)
                        {
                            vis[nx][ny] = 1;
                            q.push({nx,ny});
                        }
                    }
                }

                for(int r = min_r; r <= max_r; r++)
                {
                    for(int c = min_c; c <= max_c; c++)
                    {
                        if(s[r][c] != color)
                        {
                            ok = 0;
                            return;
                        }
                    }
                }
            }
        }
    }
}

void solve()
{
    cin >> n >> m;
    ok = 1;
    vector<string>s(n);
    vector<vector<int>>vis(n,vector<int>(m,0));
    for(int i = 0; i < n; i++) cin >> s[i];

    bfs(s ,vis);

    cout << (ok ? "YES" : "NO") << endl;
}

对每一次操作，记录所得的值的最大和最小值，dp

// 代码示例
ll calc1(ll step,vector<vector<string>>&num,vector<string>&sign,ll ccc)
{
    ll cur = ccc;
    if(num[step][0] == "x")
    {
        if(sign[step][0] == '*') cur = cur * (stoll(num[step][1]));
        else if(sign[step][0] == '-') cur = cur - (stoll(num[step][1]));
        else if(sign[step][0] == '+') cur = cur + (stoll(num[step][1]));
        else cur = cur / (stoll(num[step][1]));
    }
    else
    {
        if(sign[step][0] == '*') cur =(stoll(num[step][0])) * cur;
        else if(sign[step][0] == '-') cur = (stoll(num[step][0])) - cur;
        else if(sign[step][0] == '+') cur = cur + (stoll(num[step][0]));
        else cur = (stoll(num[step][0])) / cur;
    }
    return cur;
}

ll calc2(ll step,vector<vector<string>>&num,vector<string>&sign,ll ccc)
{
    ll cur = ccc;
    if(num[step][2] == "x")
    {
        if(sign[step][1] == '*') cur = cur * (stoll(num[step][3]));
        else if(sign[step][1] == '-') cur = cur - (stoll(num[step][3]));
        else if(sign[step][1] == '+') cur = cur + (stoll(num[step][3]));
        else cur = cur / (stoll(num[step][3]));
    }
    else
    {
        if(sign[step][1] == '*') cur =(stoll(num[step][2])) * cur;
        else if(sign[step][1] == '-') cur = (stoll(num[step][2])) - cur;
        else if(sign[step][1] == '+') cur = cur + (stoll(num[step][2]));
        else cur = (stoll(num[step][2])) / cur;
    }
    return cur;
}

void solve()
{
    ll  n,x;
    cin >> n >> x;

    vector<string>sign(n);
    vector<vector<string>>num(n);
    for(int i = 0; i < n; i++)
    {
        char sign1,sign2;
        string num1,num2,num3,num4;
        cin >> num1 >> sign1 >> num2 >> num3 >> sign2 >> num4;
        num[i].push_back(num1);
        num[i].push_back(num2);
        num[i].push_back(num3);
        num[i].push_back(num4);
        sign[i].push_back(sign1);
        sign[i].push_back(sign2);
    }
    vector<vector<ll>>dp(n,vector<ll>(2,0));
    dp[0][0]=max(calc1(0,num,sign,x),calc2(0,num,sign,x));
    dp[0][1]=min(calc1(0,num,sign,x),calc2(0,num,sign,x));
    


    for(int i = 1; i < n; i++)
    {
       dp[i][0] = max(calc1(i,num,sign,dp[i-1][0]),calc2(i,num,sign,dp[i-1][0]));
       dp[i][0] = max(dp[i][0],max(calc1(i,num,sign,dp[i-1][1]),calc2(i,num,sign,dp[i-1][1])));
       dp[i][1] = min(calc1(i,num,sign,dp[i-1][0]),calc2(i,num,sign,dp[i-1][0]));
       dp[i][1] = min(dp[i][1],min(calc1(i,num,sign,dp[i-1][1]),calc2(i,num,sign,dp[i-1][1])));
    }

    ll ans = max(dp[n-1][1],dp[n-1][0]);
    cout << ans << endl;
}

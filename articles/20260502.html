题目：牛客周赛 Round 142

B

由于每次只能选择首位和尾位，所以我们可以初始化一个按1~n的deque，对于每个值，我们检查该值是否等于deque的首元素或尾元素，如果是，则从deque中移除该元素。

// 代码示例
void solve()
{
    int n;
    cin>>n;
    deque <int> dq;

    for(int i=0;i<n;i++)
    {
        dq.push_back(i+1);
    }

    int ok=1;
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        int op=dq.front();
        int ed=dq.back();
        if(x==op) dq.pop_front();
        else if(x==ed) dq.pop_back();
        else
        {
            ok=0;
        }
    }
    cout << (ok?"YES":"NO") << endl;
}

C

注意题干的“子序列中的元素在原数组中的相对顺序必须保持不变，但不要求连续。”，所以只需按顺序逐个检查数组中的元素，初始时最长子序列长度为1，如果当前元素能被当前长度整除，则增加最长子序列长度，否则检查下一个元素。

D

“所有小朋友手上的糖数两两相等”，要达成这个条件，我们只需要确保每个小朋友最终手上的糖数都相同。所以首先判断总糖数是否能被小朋友数量整除，若不能，则可以直接判断为NO。反之，由于每个人只能给出一个糖果或获得一个糖果，所以遍历每个人，如果当前的人的糖果数大于（平均值+1）or小于（平均值-1），则可以判断为NO，如果等于平均值+1，则当前的人的糖果数-1，下一个人的糖果数+1。

// 代码示例
void solve()
{
    int n;
    cin>>n;
    vector<ll> a(n+2);
    ll sum=0;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    a[n+1]=a[1];
    a[0]=a[n];
    if(sum%n!=0)
    {
        cout<<"NO"<<endl;
        return;
    }

    ll ave=sum/n;
    for(int i=1;i<=n;i++)
    {
        if(a[i]>ave+1||a[i]<ave-1)
        {
            cout<<"NO"<<endl;
            return;
        }
        
        if(a[i]==ave+1)
        {
            if(i==n)
            {
                a[i]--;
                a[1]++;
            }
            else 
            {
                a[i]--;
                a[i+1]++;
            }
            
        }

    }
    cout<<"YES"<<endl;
    
}

E

所选区间长度>=2，所以区间内不能出现孤立一点，并且区间必须先增后减（可平）

// 代码示例
void solve()
{
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
    }
    
    if (n == 1)
    {
        if (a[0] == 0) cout << "Yes" << endl;
        else cout << "No" << endl;
        return;
    }
    
    if (a[0] > a[1] || a[n-2] < a[n-1])
    {
        cout << "No" << endl;
        return;
    }
    
    for (int i = 1; i < n - 1; i++)
    {
        if (a[i] > a[i-1] && a[i] > a[i+1])
        {
            cout << "No" << endl;
            return;
        }
    }
    
    cout << "Yes" << endl;
}
